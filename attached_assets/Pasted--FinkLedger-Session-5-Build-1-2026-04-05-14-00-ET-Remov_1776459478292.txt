// ── Parser ────────────────────────────────────────────────────────────────────

function parseOptionSymbol(raw) {
  const s = raw.trim().replace(/^-/, "");
  const m = s.match(/^([A-Z]+)(\d{6})([PC])(\d+(?:\.\d+)?)$/);
  if (!m) return null;
  const [, ticker, d, putCall, strikeStr] = m;
  return {
    ticker,
    expiration: "20" + d.slice(0,2) + "-" + d.slice(2,4) + "-" + d.slice(4,6),
    putCall,
    strike: parseFloat(strikeStr),
  };
}

function parseDate(raw) {
  const p = (raw || "").trim().split("/");
  if (p.length !== 3) return (raw || "").trim();
  return p[2] + "-" + p[0].padStart(2,"0") + "-" + p[1].padStart(2,"0");
}

function parseNum(raw) {
  if (!raw || raw.trim() === "") return 0;
  return parseFloat(raw.replace(/,/g, "")) || 0;
}

function classifyAction(a, isOption) {
  const u = a.toUpperCase();
  if (u.includes("YOU SOLD OPENING"))   return "STO";
  if (u.includes("YOU BOUGHT OPENING")) return "BTO";
  if (u.includes("YOU SOLD CLOSING"))   return "STC";
  if (u.includes("YOU BOUGHT CLOSING")) return "BTC";
  if (u.startsWith("EXPIRED") && isOption)   return "EXPIRED";
  if (u.startsWith("ASSIGNED") && isOption)  return "ASSIGNED";
  if (u.startsWith("EXERCISED") && isOption) return "EXERCISED";
  if (u.includes("YOU BOUGHT ASSIGNED PUTS"))            return "STOCK_BUY";
  if (u.includes("YOU SOLD EXERCISED PUTS"))             return "STOCK_SELL";
  if (u.includes("YOU BOUGHT EXERCISED CALLS"))          return "STOCK_BUY";
  if (u.includes("YOU SOLD SHORT SALE ASSIGNED CALLS"))  return "STOCK_SELL";
  if (u.includes("YOU SOLD ASSIGNED CALLS"))             return "STOCK_SELL";
  if (u.includes("YOU BOUGHT SHORT COVER"))              return "STOCK_COVER";
  if (u.includes("YOU SOLD RECORD DATE") && !isOption)   return "STOCK_SELL";
  if (u.includes("YOU BOUGHT RECORD DATE") && !isOption) return "STOCK_BUY";
  if (u.startsWith("YOU SOLD") && !isOption)   return "STOCK_SELL";
  if (u.startsWith("YOU BOUGHT") && !isOption) return "STOCK_BUY";
  return "UNKNOWN";
}

function parseCSV(text) {
  const lines = text.split(/\r?\n/);
  const transactions = [], skipped = [], unknown = [];
  let headerFound = false;

  for (const rawLine of lines) {
    const trimmed = rawLine.trim();
    if (/^The data and information/i.test(trimmed)) break;
    if (/^Brokerage services/i.test(trimmed)) break;
    if (!trimmed) continue;
    const cols = rawLine.split("\t");
    if (cols.length < 11) continue;
    const [runDate, action, symbol, , , priceRaw, quantityRaw, commissionRaw, feesRaw, , amountRaw] = cols;
    const dateStr   = (runDate || "").trim();
    const actionStr = (action  || "").trim();
    const symbolStr = (symbol  || "").trim();
    if (dateStr === "Run Date") { headerFound = true; continue; }
    if (!headerFound) continue;
    if (SKIP.some(function(p) { return p.test(actionStr); })) { skipped.push(actionStr); continue; }
    if (!symbolStr && !actionStr) continue;
    const isOption = symbolStr.startsWith("-") || symbolStr.startsWith(" -");
    const isStock  = !isOption && symbolStr.length >= 1 && symbolStr.length <= 6 && /^[A-Z]/.test(symbolStr);
    if (!isOption && !isStock) { if (actionStr) skipped.push(actionStr); continue; }
    let ticker, expiration = null, putCall = null, strike = null;
    if (isOption) {
      const parsed = parseOptionSymbol(symbolStr);
      if (!parsed) {
        console.warn("[FinkLedger] unrecognized option symbol:", symbolStr);
        unknown.push({ action: actionStr, symbol: symbolStr });
        continue;
      }
      ({ ticker, expiration, putCall, strike } = parsed);
    } else {
      ticker = symbolStr;
    }
    const actionType = classifyAction(actionStr, isOption);
    if (actionType === "UNKNOWN") { unknown.push({ action: actionStr, symbol: symbolStr }); continue; }
    transactions.push({
      id: crypto.randomUUID(),
      date: parseDate(dateStr),
      ticker: (ticker || "").slice(0, 12),
      action: actionType,
      isOption,
      putCall: putCall || null,
      strike: strike || null,
      expiration: expiration || null,
      quantity: Math.abs(parseNum(quantityRaw)),
      price: parseNum(priceRaw),
      amount: parseNum(amountRaw) || 0,
      commission: parseNum(commissionRaw) || 0,
      fees: parseNum(feesRaw) || 0,
      symbol: symbolStr,
      rawAction: actionStr,
    });
  }
  return { transactions, skipped, unknown };
}

// ── Chapter Grouping ──────────────────────────────────────────────────────────

function isResolved(chapter) {
  return chapter.transactions.some(function(t) {
    return t.action === "BTC" || t.action === "STC" ||
           t.action === "EXPIRED" || t.action === "ASSIGNED" || t.action === "EXERCISED";
  });
}

function buildChapters(txns) {
  const sorted = txns.slice().sort(function(a, b) {
    const dc = a.date < b.date ? -1 : a.date > b.date ? 1 : 0;
    if (dc !== 0) return dc;
    const pa = ACTION_PRIORITY[a.action] !== undefined ? ACTION_PRIORITY[a.action] : 99;
    const pb = ACTION_PRIORITY[b.action] !== undefined ? ACTION_PRIORITY[b.action] : 99;
    return pa - pb;
  });

  const chapters = [];

  for (let i = 0; i < sorted.length; i++) {
    const t = sorted[i];

    if (t.action === "STO" || t.action === "BTO") {
      let existing = null;
      for (let j = 0; j < chapters.length; j++) {
        if (chapters[j].expiration === t.expiration && !isResolved(chapters[j])) {
          existing = chapters[j]; break;
        }
      }
      if (existing) {
        existing.transactions.push(t);
      } else {
        chapters.push({ id: mkId(), expiration: t.expiration, transactions: [t], status: "OPEN", pnl: 0 });
      }

    } else if (t.action === "BTC" || t.action === "STC" ||
               t.action === "EXPIRED" || t.action === "ASSIGNED" || t.action === "EXERCISED") {
      let match = null;
      if (t.expiration) {
        for (let j = 0; j < chapters.length; j++) {
          if (chapters[j].expiration === t.expiration) { match = chapters[j]; break; }
        }
      }
      if (match) {
        match.transactions.push(t);
      } else if (chapters.length) {
        chapters[chapters.length - 1].transactions.push(t);
      }

    } else if (t.action === "STOCK_BUY" || t.action === "STOCK_SELL" || t.action === "STOCK_COVER") {
      const tTime = new Date(t.date).getTime();
      let best = null, bestDiff = Infinity;
      for (let j = 0; j < chapters.length; j++) {
        const c = chapters[j];
        const hasAssign = c.transactions.some(function(tx) {
          return tx.action === "ASSIGNED" || tx.action === "EXERCISED";
        });
        if (!hasAssign) continue;
        for (let k = 0; k < c.transactions.length; k++) {
          const tx = c.transactions[k];
          if (tx.action !== "ASSIGNED" && tx.action !== "EXERCISED") continue;
          const diff = Math.abs(new Date(tx.date).getTime() - tTime);
          if (diff < bestDiff) { bestDiff = diff; best = c; }
        }
      }
      if (best) {
        best.transactions.push(t);
      } else if (chapters.length) {
        chapters[chapters.length - 1].transactions.push(t);
      }
    }
  }

  for (let i = 0; i < chapters.length; i++) {
    const c = chapters[i];
    const acts = c.transactions.map(function(t) { return t.action; });
    c.pnl = c.transactions.reduce(function(s, t) { return s + (t.amount || 0); }, 0);

    if (acts.indexOf("ASSIGNED") !== -1 || acts.indexOf("EXERCISED") !== -1) {
      c.status = "ASSIGNED";
    } else if (acts.indexOf("EXPIRED") !== -1) {
      c.status = "EXPIRED";
    } else if (acts.indexOf("BTC") !== -1 || acts.indexOf("STC") !== -1) {
      c.status = "CLOSED";
    } else {
      c.status = "OPEN";
    }

    if (c.status === "CLOSED" && i + 1 < chapters.length) {
      const next = chapters[i + 1];
      const closeDates = c.transactions
        .filter(function(t) { return t.action === "BTC" || t.action === "STC"; })
        .map(function(t) { return t.date; }).sort();
      const openDates = next.transactions
        .filter(function(t) { return t.action === "STO" || t.action === "BTO"; })
        .map(function(t) { return t.date; }).sort();
      if (closeDates.length && openDates.length) {
        const diff = (new Date(openDates[0]).getTime() - new Date(closeDates[closeDates.length - 1]).getTime()) / 86400000;
        if (diff >= -0.5 && diff <= 1.5) c.status = "ROLLED";
      }
    }
  }

  return chapters;
}

function formatExpiry(exp) {
  if (!exp) return "";
  const parts = exp.split("-");
  const d = new Date(Number(parts[0]), Number(parts[1]) - 1, Number(parts[2]));
  return d.toLocaleDateString("en-US", { month: "short", day: "numeric", year: "numeric" });
}

function getChapterDesc(chapter) {
  const stoTx = chapter.transactions.filter(function(t) { return t.action === "STO" && t.isOption; })[0] || null;
  const btoTx = chapter.transactions.filter(function(t) { return t.action === "BTO" && t.isOption; })[0] || null;
  const anyOpt = stoTx || btoTx;
  if (!anyOpt) return { type: "Stock", strikeLabel: "", expLabel: formatExpiry(chapter.expiration) };
  const putCall = anyOpt.putCall === "P" ? "Put" : "Call";
  const ss = stoTx ? stoTx.strike : null;
  const ls = btoTx ? btoTx.strike : null;
  const strikeLabel = (ss !== null && ls !== null) ? ("$" + ss + "/$" + ls) :
                      (ss !== null) ? "$" + ss :
                      (ls !== null) ? "$" + ls : "";
  return { type: putCall, strikeLabel, expLabel: formatExpiry(chapter.expiration) };
}

// ── Data Grouping ─────────────────────────────────────────────────────────────

function groupByTicker(transactions) {
  const map = {};
  for (let i = 0; i < transactions.length; i++) {
    const t = transactions[i];
    if (!map[t.ticker]) map[t.ticker] = [];
    map[t.ticker].push(t);
  }
  return Object.keys(map).map(function(ticker) {
    const txns = map[ticker];
    const chapters = buildChapters(txns);
    const pnl = txns.reduce(function(s, t) { return s + (t.amount || 0); }, 0);
    const allDates = txns.map(function(t) { return t.date; }).sort();
    const lastDate = allDates.length ? allDates[allDates.length - 1] : "";
    const hasOpen = chapters.some(function(c) { return c.status === "OPEN"; });
    const allClosed = !hasOpen;
    return { ticker, pnl, txns, chapters, lastDate, hasOpen, allClosed };
  });
}

// ── Formatters ────────────────────────────────────────────────────────────────

function fmtDollar(n, signed) {
  const abs = Math.abs(n).toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  if (!signed) return (n < 0 ? "\u2212" : "") + "$" + abs;
  return (n < 0 ? "\u2212$" : "+$") + abs;
}

function fmtDateRange(transactions) {
  if (!transactions.length) return "";
  const dates = transactions.map(function(t) { return t.date; }).sort();
  const fmt = function(d) {
    const parts = d.split("-");
    return new Date(Number(parts[0]), Number(parts[1]) - 1).toLocaleString("en-US", { month: "short", year: "numeric" });
  };
  const first = fmt(dates[0]), last = fmt(dates[dates.length - 1]);
  return first === last ? first : first + " \u2013 " + last;
}

          value={text}
          onChange={function(e) { setText(e.target.value); setErr(""); }}
          placeholder={'[{"id":"...","date":"...","ticker":"...",...}]'}
          spellCheck={false}
          autoFocus
        />
        {err && <div style={{ color: RED, fontSize: 12, marginTop: 6 }}>{err}</div>}
        <div style={{ display: "flex", gap: 10, marginTop: 12 }}>
          <button
            onClick={onClose}
            style={{
              flex: 1, padding: "11px 0", background: "transparent",
              color: MUTED, border: "1px solid " + BORDER, borderRadius: 8,
              fontSize: 14, cursor: "pointer", fontFamily: SANS, minHeight: 44,
            }}
          >
            Cancel
          </button>
          <button
            onClick={handleRestore}
            style={{
              flex: 1, padding: "11px 0", background: TEAL_DIM,
              color: TEAL, border: "1px solid " + TEAL + "44", borderRadius: 8,
              fontSize: 14, fontWeight: 700, cursor: "pointer", fontFamily: SANS, minHeight: 44,
            }}
          >
            Restore
          </button>
        </div>
      </div>
    </div>
  );
}

// ── App ───────────────────────────────────────────────────────────────────────

export default function App() {
  const [transactions, setTransactions] = useState(function() {
    try {
      const s = localStorage.getItem(STORAGE_KEY);
      return s ? JSON.parse(s) : null;
    } catch (e) { return null; }
  });
  const [screen, setScreen]           = useState(null);
  const [csvText, setCsvText]         = useState("");
  const [parseResult, setParseResult] = useState(null);
  const [expanded, setExpanded]       = useState({});
  const [error, setError]             = useState("");
  const [sortMode, setSortMode]       = useState("alpha");
  const [filterMode, setFilterMode]   = useState("all");
  const [toasts, setToasts]           = useState([]);
  const [showHelp, setShowHelp]       = useState(false);
  const [showRestore, setShowRestore] = useState(false);
  const [confirm, setConfirm]         = useState(null);
  const [isLoading, setIsLoading]     = useState(false);

  useEffect(function() { setScreen(transactions ? "summary" : "import"); }, []);

  const groups    = useMemo(function() { return transactions ? groupByTicker(transactions) : []; }, [transactions]);
  const totalPnl  = useMemo(function() { return groups.reduce(function(s, g) { return s + g.pnl; }, 0); }, [groups]);
  const dateRange = useMemo(function() { return transactions ? fmtDateRange(transactions) : ""; }, [transactions]);
  const openCount = useMemo(function() { return groups.filter(function(g) { return g.hasOpen; }).length; }, [groups]);

  const resolvedGroups = useMemo(function() {
    return groups.filter(function(g) {
      return g.chapters.some(function(c) { return c.status !== "OPEN"; });
    });
  }, [groups]);
  const wins  = useMemo(function() { return resolvedGroups.filter(function(g) { return g.pnl > 0; }).length; }, [resolvedGroups]);
  const losses = useMemo(function() { return resolvedGroups.filter(function(g) { return g.pnl < 0; }).length; }, [resolvedGroups]);
  const best  = useMemo(function() {
    if (!groups.length) return null;
    return groups.slice().sort(function(a, b) { return b.pnl - a.pnl; })[0];
  }, [groups]);
  const worst = useMemo(function() {
    if (!groups.length) return null;
    return groups.slice().sort(function(a, b) { return a.pnl - b.pnl; })[0];
  }, [groups]);

  const displayGroups = useMemo(function() {
    let g = groups.slice();
    if (filterMode === "open")   g = g.filter(function(x) { return x.hasOpen; });
    if (filterMode === "closed") g = g.filter(function(x) { return x.allClosed; });
    if      (sortMode === "alpha")    g.sort(function(a, b) { return a.ticker < b.ticker ? -1 : a.ticker > b.ticker ? 1 : 0; });
    else if (sortMode === "pnl-desc") g.sort(function(a, b) { return b.pnl - a.pnl; });
    else if (sortMode === "pnl-asc")  g.sort(function(a, b) { return a.pnl - b.pnl; });
    else if (sortMode === "recent")   g.sort(function(a, b) { return b.lastDate < a.lastDate ? -1 : b.lastDate > a.lastDate ? 1 : 0; });
    return g;
  }, [groups, sortMode, filterMode]);

  // ── Toast helpers ──
  const addToast = useCallback(function(message, type) {
    const id = crypto.randomUUID();
    setToasts(function(prev) { return [...prev, { id, message, type: type || "success" }]; });
    setTimeout(function() {
      setToasts(function(prev) { return prev.filter(function(t) { return t.id !== id; }); });
    }, 3000);
  }, []);

  const dismissToast = useCallback(function(id) {
    setToasts(function(prev) { return prev.filter(function(t) { return t.id !== id; }); });
  }, []);

  // ── Backup ──
  const handleBackup = useCallback(function() {
    if (!transactions || !transactions.length) { addToast("Nothing to back up.", "error"); return; }
    navigator.clipboard.writeText(JSON.stringify(transactions)).then(function() {
      addToast("Backup copied to clipboard ✓");
    }).catch(function() {
      addToast("Clipboard access denied — try again.", "error");
    });
  }, [transactions, addToast]);

  // ── Restore ──
  const handleRestoreData = useCallback(function(parsed) {
    const tickerSet = new Set(parsed.map(function(t) { return t.ticker; }));
    setConfirm({
      message: "This will replace your current data with " + parsed.length + " transactions across " + tickerSet.size + " tickers. Continue?",
      onConfirm: function() {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(parsed));
        setTransactions(parsed);
        setScreen("summary");
        setShowRestore(false);
        setConfirm(null);
        addToast("Restored " + parsed.length + " transactions across " + tickerSet.size + " tickers ✓");
      },
    });
  }, [addToast]);

  // ── Parse CSV ──
  const handleParse = useCallback(function() {
    setError(""); setParseResult(null);
    if (!csvText.trim()) { setError("Paste your Fidelity activity CSV first."); return; }
    setIsLoading(true);
    setTimeout(function() {
      try {
        const result = parseCSV(csvText);
        setIsLoading(false);
        if (!result.transactions.length) {
          setError("No transactions found. Make sure you pasted the full tab-separated export.");
          return;
        }
        setParseResult(result);
      } catch (e) {
        setIsLoading(false);
        setError("Parse error: " + e.message);
      }
    }, 50);
  }, [csvText]);

  // ── Commit CSV ──
  const handleCommit = useCallback(function() {
    if (!parseResult) return;
    localStorage.setItem(STORAGE_KEY, JSON.stringify(parseResult.transactions));
    setTransactions(parseResult.transactions);
    setScreen("summary");
    setParseResult(null); setCsvText("");
    const tickers = new Set(parseResult.transactions.map(function(t) { return t.ticker; }));
    addToast("Imported " + parseResult.transactions.length + " transactions across " + tickers.size + " tickers ✓");
  }, [parseResult, addToast]);

  // ── Clear ──
  const handleClear = useCallback(function() {
    setConfirm({
      message: "Clear all data? This cannot be undone. Backup first if needed.",
      onConfirm: function() {
        localStorage.removeItem(STORAGE_KEY);
        setTransactions(null); setScreen("import");
        setParseResult(null); setCsvText(""); setExpanded({}); setError("");
        setConfirm(null);
      },
    });
  }, []);

  // ── Re-import ──
  const handleReImport = useCallback(function() {
    if (transactions) {
      setConfirm({
        message: "This will replace all existing data with the new CSV. Continue?",
        onConfirm: function() {
          setScreen("import"); setCsvText(""); setParseResult(null); setError("");
          setConfirm(null);
        },
      });
    } else {
      setScreen("import"); setCsvText(""); setParseResult(null); setError("");
    }
  }, [transactions]);

  // ── Toggle expand with scroll ──
  const toggle = useCallback(function(ticker) {
    setExpanded(function(e) {
      const n = Object.assign({}, e);
      const opening = !n[ticker];
      n[ticker] = opening;
      if (opening) {
        setTimeout(function() {
          const el = document.getElementById("ticker-" + ticker);
          if (el) el.scrollIntoView({ behavior: "smooth", block: "nearest" });
        }, 60);
      }
      return n;
    });
  }, []);

  const handleSortPnl = useCallback(function() {
    setSortMode(function(s) { return s === "pnl-desc" ? "pnl-asc" : "pnl-desc"; });
  }, []);

  if (!screen) return null;

  const iconBtn = {
    background: "rgba(255,255,255,0.08)", border: "1px solid rgba(255,255,255,0.18)",
    color: "#fff", borderRadius: 6, width: 36, height: 36,
    display: "inline-flex", alignItems: "center", justifyContent: "center",
    cursor: "pointer", fontSize: 15, flexShrink: 0, fontFamily: SANS,
    transition: "background 0.15s",
  };

  return (
    <div style={{ minHeight: "100vh", background: BG, color: HEAD, fontFamily: SANS, maxWidth: 720, margin: "0 auto", paddingBottom: 80 }}>

      {/* ── Header ── */}
      <header style={{
        background: "linear-gradient(135deg, #006978 0%, #0097a7 100%)",
        padding: "0 16px", display: "flex", alignItems: "center",
        justifyContent: "space-between", height: 56, gap: 10,
      }}>
        <div style={{ display: "flex", alignItems: "center", gap: 9, minWidth: 0 }}>
          <img
            src={FINK_IMAGE} alt=""
            style={{ width: 28, height: 28, borderRadius: "50%", border: "2px solid rgba(255,255,255,0.3)", flexShrink: 0 }}
          />
          <span style={{ fontFamily: MONO, fontWeight: 600, fontSize: 18, color: "#fff", letterSpacing: -0.5, whiteSpace: "nowrap" }}>FinkLedger</span>
          <span style={{ fontSize: 10, color: "rgba(255,255,255,0.6)", background: "rgba(0,0,0,0.2)", padding: "2px 7px", borderRadius: 10, letterSpacing: 0.5 }}>P&L</span>
        </div>
        <div style={{ display: "flex", gap: 7, alignItems: "center", flexShrink: 0 }}>
          {transactions && (
            <>
              <button className="icon-btn" onClick={handleBackup}                       style={iconBtn} title="Backup to clipboard">💾</button>
              <button className="icon-btn" onClick={function() { setShowRestore(true);}} style={iconBtn} title="Restore from backup" >📥</button>
            </>
          )}
          <button className="icon-btn" onClick={function() { setShowHelp(true); }} style={iconBtn} title="Help">ℹ️</button>
          <a
            href={FINKWATCH_URL}
            target="_blank"
            rel="noopener noreferrer"
            style={{ fontSize: 12, color: "rgba(255,255,255,0.85)", textDecoration: "none", padding: "6px 10px", border: "1px solid rgba(255,255,255,0.25)", borderRadius: 6, whiteSpace: "nowrap", height: 34, display: "flex", alignItems: "center" }}
          >
            📊 FinkWatch
          </a>
        </div>
      </header>

      {/* ── Summary Screen ── */}
      {screen === "summary" && transactions && (
        <div>
          {/* Sticky: totals + sort/filter */}
          <div style={{ position: "sticky", top: 0, zIndex: 20, background: SURFACE, borderBottom: "1px solid " + BORDER }}>
            {/* Totals bar */}
            <div style={{ padding: "10px 16px" }}>
              <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
                <div>
                  <div style={{ fontSize: 10, color: MUTED, textTransform: "uppercase", letterSpacing: 0.8, fontWeight: 600 }}>Total P&L</div>
                  <div style={{ fontSize: 11, color: DIM, marginTop: 1 }}>{dateRange}</div>
                </div>
                <div style={{ fontFamily: MONO, fontSize: 22, fontWeight: 700, color: totalPnl >= 0 ? GREEN : RED, whiteSpace: "nowrap" }}>
                  {fmtDollar(totalPnl, true)}
                </div>
              </div>

              {groups.length > 0 && (
                <div style={{
                  display: "flex", gap: 10, marginTop: 8,
                  flexWrap: "wrap", alignItems: "center",
                  paddingTop: 8, borderTop: "1px solid " + BORDER,
                  fontSize: 11, fontFamily: MONO,
                }}>
                  <span>
                    <span style={{ color: resolvedGroups.length ? (wins / resolvedGroups.length >= 0.5 ? GREEN : RED) : MUTED, fontWeight: 700 }}>
                      {resolvedGroups.length ? Math.round(wins / resolvedGroups.length * 100) + "%" : "—"}
                    </span>
                    <span style={{ color: DIM }}>{" win"}</span>
                  </span>
                  <span style={{ color: DIM }}>|</span>
                  <span>
                    <span style={{ color: GREEN, fontWeight: 600 }}>{wins + "W"}</span>
                    <span style={{ color: DIM }}>{" / "}</span>
                    <span style={{ color: RED, fontWeight: 600 }}>{losses + "L"}</span>
                  </span>
                  {best && (
                    <span style={{ color: DIM }}>
                      {"Best: "}<span style={{ color: GREEN }}>{best.ticker + " " + fmtDollar(best.pnl, true)}</span>
                    </span>
                  )}
                  {worst && worst.ticker !== (best && best.ticker) && (
                    <span style={{ color: DIM }}>
                      {"Worst: "}<span style={{ color: worst.pnl < 0 ? RED : BODY_CLR }}>{worst.ticker + " " + fmtDollar(worst.pnl, true)}</span>
                    </span>
                  )}
                </div>
              )}
            </div>

            {/* Sort + Filter pills */}
            <div style={{
              display: "flex", alignItems: "center", gap: 8,
              padding: "8px 12px", borderTop: "1px solid " + BORDER,
              flexWrap: "nowrap", overflowX: "auto",
            }}>
              <Pill label="All"    active={filterMode === "all"}    onClick={function() { setFilterMode("all"); }} />
              <Pill label={"Open" + (openCount ? " (" + openCount + ")" : "")} active={filterMode === "open"} onClick={function() { setFilterMode("open"); }} />
              <Pill label="Closed" active={filterMode === "closed"} onClick={function() { setFilterMode("closed"); }} />
              <div style={{ width: 1, height: 18, background: BORDER, flexShrink: 0 }} />
              <Pill label={"A\u2013Z"} active={sortMode === "alpha"}  onClick={function() { setSortMode("alpha"); }} />
              <Pill
                label={sortMode === "pnl-asc" ? "P&L \u2191" : "P&L \u2193"}
                active={sortMode === "pnl-desc" || sortMode === "pnl-asc"}
                onClick={handleSortPnl}
              />
              <Pill label="Recent" active={sortMode === "recent"}   onClick={function() { setSortMode("recent"); }} />
            </div>
          </div>

          {/* Ticker list */}
          <div style={{ padding: "10px 12px" }}>
            {displayGroups.length === 0 && (
              <div style={{ textAlign: "center", padding: "48px 20px", color: MUTED, fontSize: 13 }}>
                {filterMode === "open"   ? "No open positions." :
                 filterMode === "closed" ? "No closed positions." :
                 "No positions to display."}
              </div>
            )}

            {displayGroups.map(function(group) {
              const { ticker, pnl, chapters, hasOpen } = group;
              const isExpanded = !!expanded[ticker];
              return (
                <div
                  key={ticker}
                  id={"ticker-" + ticker}
                  style={{
                    marginBottom: 6, borderRadius: 8, overflow: "hidden",
                    border: "1px solid " + BORDER,
                    borderLeft: "3px solid " + (hasOpen ? TEAL : DIM),
                  }}
                >
                  {/* Ticker row */}
                  <div
                    className="ticker-row"
                    style={{
                      display: "flex", justifyContent: "space-between", alignItems: "center",
                      padding: "12px 14px", background: SURFACE, cursor: "pointer",
                      userSelect: "none", minHeight: 52,
                    }}
                    onClick={function() { toggle(ticker); }}
                    role="button" tabIndex={0}
                    onKeyDown={function(e) { if (e.key === "Enter" || e.key === " ") toggle(ticker); }}
                  >
                    <div style={{ display: "flex", alignItems: "center", gap: 10, minWidth: 0 }}>
                      <span style={{
                        fontSize: 9, color: MUTED, flexShrink: 0,
                        display: "inline-block", transition: "transform 0.15s",
                        transform: isExpanded ? "rotate(90deg)" : "rotate(0deg)",
                      }}>&#9654;</span>
                      <span style={{ fontWeight: 700, fontSize: 15, color: HEAD, whiteSpace: "nowrap" }}>{ticker}</span>
                      <span style={{
                        fontSize: 10, fontFamily: MONO, color: MUTED,
                        background: SURFACE2, border: "1px solid " + BORDER,
                        padding: "2px 7px", borderRadius: 10, whiteSpace: "nowrap",
                      }}>
                        {chapters.length + " trade" + (chapters.length !== 1 ? "s" : "")}
                      </span>
                      <span style={{
                        width: 6, height: 6, borderRadius: "50%",
                        background: hasOpen ? TEAL : DIM, display: "inline-block", flexShrink: 0,
                        boxShadow: hasOpen ? "0 0 5px " + TEAL : "none",
                      }} />
                    </div>
                    <span style={{
                      fontFamily: MONO, fontSize: 16, fontWeight: 700,
                      color: pnl >= 0 ? GREEN : RED, whiteSpace: "nowrap", marginLeft: 12,
                    }}>
                      {fmtDollar(pnl, true)}
                    </span>
                  </div>

                  {/* Expanded chapters */}
                  {isExpanded && (
                    <div style={{ background: BG, animation: "expandIn 0.15s ease" }}>
                      {chapters.map(function(chapter, ci) {
                        const desc       = getChapterDesc(chapter);
                        const statusMeta = CHAPTER_STATUS_META[chapter.status] || CHAPTER_STATUS_META.OPEN;
                        const showRollConnector = ci > 0 && chapters[ci - 1].status === "ROLLED";

                        // Opening credit / debit
                        const openingAmt = chapter.transactions
                          .filter(function(t) { return t.action === "STO" || t.action === "BTO"; })
                          .reduce(function(s, t) { return s + (t.amount || 0); }, 0);
                        const openingLabel = openingAmt !== 0
                          ? "opened for " + fmtDollar(Math.abs(openingAmt)) + (openingAmt > 0 ? " credit" : " debit")
                          : null;

                        return (
                          <div key={chapter.id}>
                            {showRollConnector && (
                              <div style={{
                                display: "flex", alignItems: "center", gap: 10,
                                padding: "5px 14px", background: "#a78bfa0d",
                                borderTop: "1px solid " + BORDER,
                              }}>
                                <div style={{ flex: 1, height: 1, background: "#a78bfa35" }} />
                                <span style={{ fontFamily: MONO, fontSize: 10, color: VIOLET, fontWeight: 600, letterSpacing: 0.3, whiteSpace: "nowrap" }}>
                                  {"\u2193 rolled \u2192"}
                                </span>
                                <div style={{ flex: 1, height: 1, background: "#a78bfa35" }} />
                              </div>
                            )}

                            {/* Chapter header */}
                            <div style={{
                              display: "flex", justifyContent: "space-between", alignItems: "flex-start",
                              padding: "9px 14px", background: SURFACE2,
                              borderTop: "1px solid " + BORDER, gap: 10,
                            }}>
                              <div style={{ display: "flex", flexDirection: "column", gap: 4, minWidth: 0, flex: 1 }}>
                                <div style={{ display: "flex", alignItems: "center", gap: 7, flexWrap: "wrap" }}>
                                  <span style={{ fontSize: 13, fontWeight: 600, color: HEAD, whiteSpace: "nowrap" }}>
                                    {desc.type + (desc.strikeLabel ? " " + desc.strikeLabel : "")}
                                  </span>
                                  {desc.expLabel && (
                                    <span style={{ fontSize: 11, color: MUTED, whiteSpace: "nowrap" }}>
                                      {"\u00b7 " + desc.expLabel}
                                    </span>
                                  )}

                                    fontSize: 10, fontWeight: 700,
                                    padding: "1px 7px", borderRadius: 10,
                                    color: statusMeta.color, background: statusMeta.bg,
                                    border: "1px solid " + statusMeta.color + "30",
                                    whiteSpace: "nowrap",

                                    {statusMeta.label}
                                  </span>
                                </div>
                                {openingLabel && (
                                  <span style={{
                                    fontSize: 10, fontFamily: MONO,
                                    color: openingAmt > 0 ? GREEN : RED,
                                  }}>
                                    {openingLabel}
                                  </span>
                                )}
                              </div>
                              <span style={{
                                fontFamily: MONO, fontSize: 13, fontWeight: 700,
                                color: chapter.pnl >= 0 ? GREEN : RED,
                                whiteSpace: "nowrap", flexShrink: 0, marginTop: 2,
                              }}>
                                {fmtDollar(chapter.pnl, true)}
                              </span>
                            </div>

                            {/* Transaction lines */}
                            {chapter.transactions.map(function(t) {
                              const meta = ACTION_META[t.action] || { label: t.action, color: MUTED, bg: DIM + "15" };
                              return (
                                <div
                                  key={t.id}
                                  style={{
                                    display: "grid",
                                    gridTemplateColumns: "88px 46px 1fr 90px",
                                    alignItems: "center", gap: 8,
                                    padding: "7px 14px 7px 24px",
                                    borderTop: "1px solid " + BORDER,
                                  }}
                                >
                                  <span style={{ fontFamily: MONO, fontSize: 11, color: MUTED }}>{t.date}</span>
                                  <span style={{
                                    fontFamily: MONO, fontSize: 10, fontWeight: 700,
                                    padding: "2px 4px", borderRadius: 4, textAlign: "center",
                                    color: meta.color, background: meta.bg,
                                  }}>
                                    {meta.label}
                                  </span>
                                  <span style={{ fontSize: 11, color: MUTED, overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap" }}>
                                    {t.isOption
                                      ? (t.putCall === "P" ? "Put" : "Call") + " $" + t.strike + " \u00b7 " + t.expiration
                                      : "Stock"}
                                  </span>
                                  <span style={{
                                    fontFamily: MONO, fontSize: 12, fontWeight: 600, textAlign: "right",
                                    color: t.amount === 0 ? DIM : t.amount > 0 ? GREEN : RED,
                                    whiteSpace: "nowrap",
                                  }}>
                                    {t.amount === 0 ? "$0.00" : fmtDollar(t.amount, true)}
                                  </span>
                                </div>
                              );
                            })}
                          </div>
                        );
                      })}

                      {/* Subtotal footer */}
                      <div style={{
                        display: "flex", justifyContent: "space-between",
                        padding: "8px 14px", borderTop: "1px solid " + BORDER,
                        background: SURFACE, fontFamily: MONO, fontSize: 12,
                      }}>
                        <span style={{ color: DIM }}>Subtotal</span>
                        <span style={{ fontWeight: 700, color: pnl >= 0 ? GREEN : RED }}>{fmtDollar(pnl, true)}</span>
                      </div>
                    </div>
                  )}
                </div>
              );
            })}
          </div>

          {/* Footer actions */}
          <div style={{
            display: "flex", justifyContent: "flex-end", gap: 10,
            padding: "14px 16px", borderTop: "1px solid " + BORDER,
          }}>
            <button
              style={{
                padding: "10px 16px", background: "transparent", color: BODY_CLR,
                border: "1px solid " + BORDER, borderRadius: 6, fontSize: 13,
                cursor: "pointer", minHeight: 44, fontFamily: SANS,
              }}
              onClick={handleReImport}
            >
              Re-import
            </button>
            <button
              style={{
                padding: "10px 16px", background: "transparent", color: RED,
                border: "1px solid #450a0a", borderRadius: 6, fontSize: 13,
                cursor: "pointer", minHeight: 44, fontFamily: SANS,
              }}
              onClick={handleClear}
            >
              Clear Data
            </button>
          </div>
        </div>
      )}

      {/* ── Import Screen ── */}
      {screen === "import" && (
        <div style={{ padding: 24 }}>
          <div style={{ textAlign: "center", marginBottom: 24 }}>
            <img
              src={FINK_IMAGE} alt=""
              style={{ width: 56, height: 56, borderRadius: "50%", border: "3px solid " + TEAL, marginBottom: 12 }}
            />
            <div style={{ fontFamily: MONO, fontSize: 20, fontWeight: 600, color: HEAD, marginBottom: 4 }}>FinkLedger</div>
            <div style={{ fontSize: 13, color: MUTED }}>Paste your Fidelity Activity History CSV to get started</div>
            <div style={{ fontSize: 12, color: DIM, marginTop: 4 }}>
              Accounts \u2192 Activity &amp; Orders \u2192 History \u2192 Export CSV
            </div>
          </div>

          <textarea
            style={{
              width: "100%", height: 200, background: SURFACE, border: "1px solid " + BORDER,
              borderRadius: 8, color: BODY_CLR, fontSize: 11, padding: "12px 14px",
              resize: "vertical", outline: "none", lineHeight: 1.5,
            }}
            value={csvText}
            onChange={function(e) { setCsvText(e.target.value); setParseResult(null); setError(""); }}
            placeholder={"Run Date\tAction\tSymbol\t..."}
            spellCheck={false}
          />

          {error && <div style={{ color: RED, fontSize: 13, marginTop: 8 }}>{error}</div>}

          <button
            style={{
              marginTop: 12, width: "100%", padding: "13px 0",
              background: isLoading ? DIM : TEAL,
              color: "#fff", border: "none", borderRadius: 8,
              fontSize: 15, fontWeight: 700,
              cursor: isLoading ? "not-allowed" : "pointer",
              minHeight: 50, fontFamily: SANS,
            }}
            onClick={handleParse}
            disabled={isLoading}
          >
            {isLoading ? "Parsing\u2026" : "Parse CSV"}
          </button>

          {parseResult && (
            <div style={{ marginTop: 16, background: SURFACE, border: "1px solid " + BORDER, borderRadius: 8, padding: 16 }}>
              <div style={{ display: "flex", gap: 16, fontSize: 13, flexWrap: "wrap", marginBottom: 10 }}>
                <span style={{ color: GREEN }}>{"✓ " + parseResult.transactions.length + " transactions"}</span>
                <span style={{ color: MUTED }}>{parseResult.skipped.length + " filtered"}</span>
                {parseResult.unknown.length > 0 && (
                  <span style={{ color: ORANGE }}>{"⚠ " + parseResult.unknown.length + " unrecognized"}</span>
                )}
              </div>
              {parseResult.unknown.length > 0 && (
                <details style={{ marginBottom: 12 }}>
                  <summary style={{ fontSize: 12, color: ORANGE, cursor: "pointer" }}>Show unrecognized rows</summary>
                  <div style={{ marginTop: 6, display: "flex", flexDirection: "column", gap: 2 }}>
                    {parseResult.unknown.map(function(u, i) {
                      return (
                        <div key={i} style={{ display: "grid", gridTemplateColumns: "60px 1fr", gap: 8, fontFamily: MONO, fontSize: 11, color: MUTED }}>
                          <span>{u.symbol || "\u2014"}</span>
                          <span>{u.action.slice(0, 80) + (u.action.length > 80 ? "\u2026" : "")}</span>
                        </div>
                      );
                    })}
                  </div>
                </details>
              )}
              <button
                style={{
                  width: "100%", padding: "13px 0", background: "#052e16",
                  color: GREEN, border: "1px solid #14532d", borderRadius: 8,
                  fontSize: 14, fontWeight: 700, cursor: "pointer",
                  minHeight: 50, fontFamily: SANS,
                }}
                onClick={handleCommit}
              >
                {"Load " + parseResult.transactions.length + " transactions \u2192"}
              </button>
            </div>
          )}

          {/* Back to summary if data exists */}
          {transactions && (
            <button
              style={{
                marginTop: 12, width: "100%", padding: "11px 0",
                background: "transparent", color: MUTED,
                border: "1px solid " + BORDER, borderRadius: 8,
                fontSize: 14, cursor: "pointer", minHeight: 46, fontFamily: SANS,
              }}
              onClick={function() { setScreen("summary"); setCsvText(""); setParseResult(null); setError(""); }}
            >
              \u2190 Back to Summary
            </button>
          )}
        </div>
      )}

      {/* ── Overlays ── */}
      {showHelp    && <HelpModal onClose={function() { setShowHelp(false); }} />}
      {showRestore && <RestoreModal onClose={function() { setShowRestore(false); }} onRestore={handleRestoreData} />}
      {confirm && (
        <ConfirmModal
          message={confirm.message}
          onConfirm={confirm.onConfirm}
          onCancel={function() { setConfirm(null); }}
        />
      )}
      <ToastContainer toasts={toasts} onDismiss={dismissToast} />
    </div>
  );
}
